120std::vector<std::vector<T>> generateSchedule(const std::vector<unsigned long long>& subscribers_periods, const std::vector<T>& subscribers_information)
121{
122 unsigned long long hyper_period = computeLeastCommonMultiple(subscribers_periods);
123 std::vector<std::vector<T>> generated_schedule(hyper_period);
124
125 for (size_t iterator = 0; iterator < subscribers_periods.size(); ++iterator)
126 {
127 auto insertIfNotPresent = [&](size_t time_index, const T& info)
128 {
129 auto& slot = generated_schedule[time_index];
130 if (std::find(slot.begin(), slot.end(), info) == slot.end())
131 {
132 slot.push_back(info);
133 }
134 };
135
136 insertIfNotPresent(0, subscribers_information[iterator]);
137 unsigned long long time = subscribers_periods[iterator];
138 while (time < hyper_period)
139 {
140 insertIfNotPresent(time, subscribers_information[iterator]);
141 time += subscribers_periods[iterator];
142 }
143 }
144
145 return generated_schedule;
146}
